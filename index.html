const canvas = document.getElementById("gameCanvas"); const ctx = canvas.getContext("2d"); let x = 100, y = 100; // Ball starting position function gameLoop() { const gamepads = navigator.getGamepads(); if (gamepads[0]) { const gp = gamepads[0]; // gp.axes[0] is Left Stick X, gp.axes[1] is Left Stick Y x += gp.axes[0] * 5; y += gp.axes[1] * 5; } // Draw the ball ctx.clearRect(0, 0, canvas.width, canvas.height); ctx.beginPath(); ctx.arc(x, y, 20, 0, Math.PI * 2); ctx.fillStyle = "blue"; ctx.fill(); requestAnimationFrame(gameLoop); } window.addEventListener("gamepadconnected", () => { console.log("Gamepad connected!"); gameLoop(); });
